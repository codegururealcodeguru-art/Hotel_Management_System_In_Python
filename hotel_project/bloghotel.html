Source Code

import tkinter as tk
from tkinter import messagebox

class HotelManagement:
    def __init__(self, root):
        self.root = root
        self.root.title("Hotel Management System")
        self.root.geometry("500x600")
        self.root.config(bg="lightblue")

        self.guests = []

        tk.Label(root, text="Hotel Check-In System",
                 font=("Arial", 18, "bold"),
                 bg="skyblue").pack(pady=10)

        self.name_var = tk.StringVar()
        self.phone_var = tk.StringVar()
        self.room_var = tk.StringVar()

        tk.Label(root, text="Guest Name:", bg="lightblue").pack()
        tk.Entry(root, textvariable=self.name_var).pack()

        tk.Label(root, text="Phone Number:", bg="lightblue").pack()
        tk.Entry(root, textvariable=self.phone_var).pack()

        tk.Label(root, text="Room Type:", bg="lightblue").pack()
        tk.OptionMenu(root, self.room_var,
                      "Single",
                      "Double",
                      "Deluxe",
                      "Suite").pack()

        tk.Button(root, text="Check-In",
                  command=self.check_in,
                  bg="green", fg="white").pack(pady=10)

        self.output = tk.Text(root, height=12, width=60)
        self.output.pack()

    def check_in(self):
        name = self.name_var.get()
        phone = self.phone_var.get()
        room = self.room_var.get()

        if name and phone:
            self.guests.append({
                "Name": name,
                "Phone": phone,
                "Room": room
            })
            messagebox.showinfo("Success", "Guest Checked In")
        else:
            messagebox.showwarning("Error", "Fill all fields")


root = tk.Tk()
app = HotelManagement(root)
root.mainloop()
